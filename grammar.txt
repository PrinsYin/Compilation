PROGRAM -> M VARS
M -> none
VARS -> VAR
VARS -> VAR VARS
VAR -> INT ID VARTYPE ;
VAR -> FUNC SENBLOCK
VARTYPE -> none
VARTYPE -> ARRAYVAR
FUNC -> VOID ID M ( PARAM )
FUNC -> INT ID M ( PARAM )
ARRAYVAR -> [ NUM ]
ARRAYVAR -> [ NUM ] ARRAYVAR
PARAM -> PARAMLIST
PARAM -> VOID
PARAM -> none
PARAMLIST -> INT ID
PARAMLIST -> INT ID , PARAMLIST
SENBLOCK -> { INASSERTION SENSEQ }
INASSERTION -> INVARDEF ; INASSERTION
INASSERTION -> none
INVARDEF -> INT ID
INVARDEF -> INT ID ARRAYVAR
SENSEQ -> SENTENCE
SENSEQ -> SENTENCE SENSEQ
SENTENCE -> IFSEN
SENTENCE -> WHILESEN
SENTENCE -> RETSEN ;
SENTENCE -> ASSIGNMENT ;
ASSIGNMENT -> ID ASSIGN EXPRESSION
ASSIGNMENT -> ARRAY ASSIGN EXPRESSION
RETSEN -> RET EXPRESSION
RETSEN -> RET
WHILESEN -> B WHILE ( CTRL ) T SENBLOCK
B -> none
IFSEN -> IF ( CTRL ) T SENBLOCK
IFSEN -> IF ( CTRL ) T SENBLOCK ELSE N SENBLOCK
CTRL -> EXPRESSION
T -> none
N -> none
EXPRESSION -> BAND
EXPRESSION -> BAND OR EXPRESSION
BAND -> BNOT
BAND -> BNOT AND BAND
BNOT -> COMP
BNOT -> NOT COMP
COMP -> PLUSEX
COMP -> PLUSEX CMP COMP
PLUSEX -> TERM
PLUSEX -> TERM OP1 PLUSEX
TERM -> FACTOR
TERM -> FACTOR OP2 TERM
FACTOR -> NUM
FACTOR -> ( EXPRESSION )
FACTOR -> ID
FACTOR -> ARRAY
FACTOR -> ID CALL
FACTOR -> ( ASSIGNMENT )
CALL -> ( ACTUALPARAM )
ARRAY -> ID [ EXPRESSION ]
ARRAY -> ARRAY [ EXPRESSION ]
ACTUALPARAM -> ACTUALPARAMLIST
ACTUALPARAM -> none
ACTUALPARAMLIST -> EXPRESSION
ACTUALPARAMLIST -> EXPRESSION , ACTUALPARAMLIST
